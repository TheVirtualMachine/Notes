\section{Differential Equations}
\paragraph{Separable differential equations}
	\begin{align}
		\frac{\dif y}{\dif x} &= \frac{g(x)}{h(y)}\\
		h(y) \dif y &= g(x) \dif x\\
		\int h(y) \dif y = \int g(x) \dif x
	\end{align}
\paragraph{Exponential growth and decay}
	\begin{align}
		y' &= ky\\
		\frac{\dif y}{\dif x} &= ky\\
		\int \frac{\dif y}{y} &= \int k \dif x\\
		\ln\abs{y} &= kx + C\\
		y &= \pm e^{kx + C}\\
		y &= \pm Ae^{kx}
	\end{align}
